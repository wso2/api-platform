Next Steps:

1. Verify the operator is running:
   $ kubectl get deployment {{ include "gateway-operator.fullname" . }} -n {{ .Release.Namespace }}
   $ kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}

2. Check operator logs:
   $ kubectl logs -f deployment/{{ include "gateway-operator.fullname" . }} -n {{ .Release.Namespace }}

3. Create an APIGateway to deploy a gateway instance:
   $ kubectl apply -f - <<EOF
   apiVersion: gateway.api-platform.wso2.com/v1alpha1
   kind: APIGateway
   metadata:
     name: my-gateway
     namespace: default
   spec:
     apiSelector:
       scope: Cluster
     storage:
       type: sqlite
     # configRef: { name: my-gateway-values }  # ConfigMap with gateway Helm values
   EOF

4. Create a RestApi to deploy an API:
   $ kubectl apply -f - <<EOF
   apiVersion: gateway.api-platform.wso2.com/v1alpha1
   kind: RestApi
   metadata:
     name: my-api
     namespace: default
   spec:
     displayName: my-api
     version: v1.0
     context: /my-api
     upstream:
       main:
         url: https://httpbin.org/anything
     operations:
       - method: GET
         path: /
   EOF

5. Monitor the resources:
   $ kubectl get apigateways -A
   $ kubectl get restapis -A
