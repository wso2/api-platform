To access the Event Gateway Runtime:
  {{- if eq .Values.eventGateway.gatewayRuntime.service.type "LoadBalancer" }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status by running:
    $ kubectl get svc {{ include "event-gateway.fullname" . }}-gateway-runtime -n {{ .Release.Namespace }} -w

  Once available, get the external IP:
    $ export SERVICE_IP=$(kubectl get svc {{ include "event-gateway.fullname" . }}-gateway-runtime -n {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
    $ echo http://$SERVICE_IP:{{ .Values.eventGateway.gatewayRuntime.service.ports.websubHttp }}
  {{- else if eq .Values.eventGateway.gatewayRuntime.service.type "NodePort" }}
    $ export NODE_PORT=$(kubectl get svc {{ include "event-gateway.fullname" . }}-gateway-runtime -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
    $ export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
    $ echo http://$NODE_IP:$NODE_PORT
  {{- else }}
    $ kubectl port-forward svc/{{ include "event-gateway.fullname" . }}-gateway-runtime {{ .Values.eventGateway.gatewayRuntime.service.ports.websubHttp }}:{{ .Values.eventGateway.gatewayRuntime.service.ports.websubHttp }} -n {{ .Release.Namespace }}

  Then visit: http://localhost:{{ .Values.eventGateway.gatewayRuntime.service.ports.websubHttp }}
  {{- end }}
{{- end }}
